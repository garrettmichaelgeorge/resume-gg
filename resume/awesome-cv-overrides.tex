% Redefine an entry of cv information
% Usage: \cventry{<position>}{<title>}{<location>}{<date>}{<description>}
\renewcommand*{\cventry}[5]{%
	% \vspace{-2.0mm}
	\setlength\tabcolsep{0pt}
	% \setlength{\extrarowheight}{0pt}

	\begingroup
	\raggedright
	\ifempty{#2#3}
	{%
		\entrypositionstyle{#1} \\
		\entrydatestyle{#4} \\
	}{%
		\entrytitlestyle{#1} \\
		\entrypositionstyle{#2} \\
		\entrydatestyle{#4} | \entrylocationstyle{#3} \\
		\vspace{1em}
	}
	\ifstrempty{#5}
	{}
	{\descriptionstyle{#5}}
	\endgroup
}
